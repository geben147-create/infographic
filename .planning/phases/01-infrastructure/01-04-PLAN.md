    Pattern:
    ```python
    import pytest

    from temporalio.worker import Worker
    from temporalio.contrib.pydantic import pydantic_data_converter
    from src.workflows.pipeline_validation import PipelineValidationWorkflow, ValidationParams
    from src.activities.stubs import stub_gpu_activity, stub_cpu_activity

        await env.shutdown()

    async def test_pipeline_validation_workflow(temporal_env, tmp_path):
        async with Worker(
            temporal_env.client,
            task_queue="gpu-queue",
            workflows=[PipelineValidationWorkflow],
            activities=[stub_gpu_activity],
        ) as gpu_worker, Worker(
            temporal_env.client,
            task_queue="cpu-queue",
            workflows=[],

        ) as cpu_worker:
            result = await temporal_env.client.execute_workflow(
    ```
