2. Create `tests/test_download_endpoints.py`:
   Use `tmp_path` fixture and `unittest.mock.patch` to redirect the file path lookup in the download endpoints to a temp directory, avoiding file creation in system directories.

   ```python
   """Tests for GET /api/pipeline/{id}/download and /api/pipeline/{id}/thumbnail."""
   import pathlib
   from unittest.mock import patch

   import pytest
   from fastapi import FastAPI
   from fastapi.testclient import TestClient

   from src.api.pipeline import router


   def _make_app():
       app = FastAPI()
       app.include_router(router)
       return app


   class TestDownloadVideo:
       def test_download_404_when_missing(self):
           app = _make_app()
           client = TestClient(app)
           resp = client.get("/api/pipeline/nonexistent-run/download")
           assert resp.status_code == 404

       def test_download_200_when_file_exists(self, tmp_path):
           workflow_id = "test-download-run"
           video_dir = tmp_path / workflow_id / "final"
           video_dir.mkdir(parents=True)
           video_file = video_dir / "output.mp4"
           video_file.write_bytes(b"\x00" * 100)  # Fake video bytes

           app = _make_app()
           client = TestClient(app)

           # Patch the path construction in the download endpoint to use tmp_path
           with patch(
               "src.api.pipeline.pathlib.Path",
               return_value=tmp_path,
           ) as mock_path_cls:
               # We need to patch so that pathlib.Path("data/pipeline") / wf_id / "final" / "output.mp4"
               # resolves to our tmp_path location.
               # Simpler approach: patch at a higher level by making the constructed path point to tmp_path
               pass

           # Most reliable approach: create file at expected path, test, then clean up
           expected = pathlib.Path("data/pipeline") / workflow_id / "final"
           expected.mkdir(parents=True, exist_ok=True)
           (expected / "output.mp4").write_bytes(b"\x00" * 100)

           try:
               resp = client.get(f"/api/pipeline/{workflow_id}/download")
               assert resp.status_code == 200
               assert resp.headers["content-type"] == "video/mp4"
               assert "attachment" in resp.headers.get("content-disposition", "")
           finally:
               import shutil
               shutil.rmtree(pathlib.Path("data/pipeline") / workflow_id, ignore_errors=True)


   class TestDownloadThumbnail:
       def test_thumbnail_404_when_missing(self):
           app = _make_app()
           client = TestClient(app)
           resp = client.get("/api/pipeline/nonexistent-run/thumbnail")
           assert resp.status_code == 404

       def test_thumbnail_200_when_file_exists(self, tmp_path):
           workflow_id = "test-thumb-run"
           expected = pathlib.Path("data/pipeline") / workflow_id / "thumbnails"
           expected.mkdir(parents=True, exist_ok=True)
           (expected / "thumbnail.jpg").write_bytes(b"\xff\xd8\xff" + b"\x00" * 100)

           app = _make_app()
           client = TestClient(app)

           try:
               resp = client.get(f"/api/pipeline/{workflow_id}/thumbnail")
               assert resp.status_code == 200
               assert resp.headers["content-type"] == "image/jpeg"
               assert "attachment" in resp.headers.get("content-disposition", "")
           finally:
               import shutil
               shutil.rmtree(pathlib.Path("data/pipeline") / workflow_id, ignore_errors=True)
   ```

   Note: The `test_download_200_when_file_exists` test creates files under `data/pipeline/` (relative to CWD) for the 200-case tests, then cleans them up in a `finally` block. The `tmp_path` fixture is accepted as a parameter for pytest compatibility. A future improvement would be to refactor the download endpoints to accept a configurable base path, enabling pure `tmp_path` testing.
